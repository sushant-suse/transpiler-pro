39def load_config(config_path: Path) -> Dict[str, Any]: 

40 """ 

41 Loads global pipeline settings from a TOML configuration file. 

42  

43 Args: 

44 config_path: Path to the `pyproject.toml` file. 

45  

46 Returns: 

47 A dictionary containing the `tool.transpiler-pro` configuration block. 

48 """ 

49 if not config_path.exists(): 

50 return {} 

51 try: 

52 with open(config_path, "rb") as f: 

53 return tomllib.load(f).get("tool", {}).get("transpiler-pro", {}) 

54 except Exception: 

55 return {}

57def sync_styles() -> None: 

58 """ 

59 Synchronizes the SUSE Style Guide submodule with the remote repository. 

60  

61 This ensures that the latest enterprise linguistic standards are applied  

62 before the linting phase begins. 

63 """ 

64 console.print("\n[bold blue]Pre-flight:[/] Syncing SUSE Style Guide via Git...") 

65 try: 

66 subprocess.run( 

67 ["git", "submodule", "update", "--init", "--remote", "styles/suse-styles"], 

68 check=True, 

69 capture_output=True, 

70 text=True 

71 ) 

72 console.print(" [bold green]✓[/] Style guide is synchronized and up-to-date.") 

73 except subprocess.CalledProcessError as e: 

74 console.print(f" [bold yellow]⚠️ Warning:[/] Sync failed. Using local cached styles. {e.stderr}") 

75 except FileNotFoundError: 

76 console.print(" [bold red]Error:[/] Git not found in system path. Skipping sync.")

78def run_pipeline( 

79 file_name: Optional[str] = None, 

80 lint_only: bool = False, 

81 fix: bool = False, 

82 sync: bool = False, 

83 config_path: Path = DEFAULT_CONFIG 

84) -> None: 

85 """ 

86 The decoupled logic for the transformation and validation pipeline. 

87  

88 This function allows for direct invocation without CLI-specific  

89 argument validation, facilitating more robust testing. 

90 """ 

91 # 0. Optional Style Sync 

92 if sync: 

93 sync_styles() 

94 

95 # 1. Load configuration and inject into engines 

96 pipeline_config = load_config(config_path) 

97 converter = DocConverter(config_path=config_path) 

98 fixer = StyleFixer(config_path=config_path) 

99 

100 # 2. Target Detection 

101 if file_name: 

102 target_files = [INPUT_DIR / file_name] 

103 else: 

104 exts = pipeline_config.get("pipeline", {}).get("supported_extensions", [".md", ".mdx"]) 

105 if not INPUT_DIR.exists(): 

106 console.print(f"[bold yellow]Notification:[/] Directory {INPUT_DIR} not found.") 

107 return 

108 target_files = [p for p in INPUT_DIR.iterdir() if p.suffix in exts] 

109 

110 if not target_files: 

111 console.print("[bold yellow]Notification:[/] No source files detected.") 

112 return 

113 

114 # 3. Pipeline Execution Loop 

115 for md_path in target_files: 

116 if not md_path.exists(): 

117 continue 

118 

119 adoc_path = OUTPUT_DIR / md_path.with_suffix(".adoc").name 

120 

121 # Phase 1: Conversion 

122 if not lint_only: 

123 console.print(f"\n[bold blue]Phase 1:[/] Converting [cyan]{md_path.name}[/]") 

124 try: 

125 converter.convert_file(md_path, adoc_path) 

126 except Exception as e: 

127 console.print(f" [bold red]Error:[/] {e}") 

128 continue 

129 

130 # Phase 2 & 3: Linter & Fixer 

131 if adoc_path.exists(): 

132 console.print(f"[bold blue]Phase 2:[/] Validating [cyan]{adoc_path.name}[/]") 

133 linter = StyleLinter(adoc_path, config_path=config_path) 

134 linter.setup_config() 

135 findings = linter.run() 

136 linter.display_report(findings) 

137 

138 if fix and findings: 

139 console.print(f"[bold blue]Phase 3:[/] Auto-repairing [cyan]{adoc_path.name}[/]") 

140 file_key = str(adoc_path.resolve()) 

141 file_violations = findings.get(file_key, []) 

142 

143 if file_violations: 

144 repaired_count = fixer.fix_file(adoc_path, file_violations) 

145 if repaired_count > 0: 

146 console.print(f" [bold green]✨ Repaired {repaired_count} violations.[/]") 

147 linter.display_report(linter.run())
