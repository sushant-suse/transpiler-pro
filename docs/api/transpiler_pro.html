Location: src/transpiler_pro/__init__.py. Description: Transpiler-Pro - Enterprise Documentation Pipeline.

This package provides a comprehensive engine for converting Markdown to AsciiDoc while enforcing linguistic standards and branding consistency.

Core Features:

  1. DocConverter: Shield-Transpile-Restore logic using Pandoc.
  2. StyleLinter: Orchestration of Vale-based style validation.
  3. StyleFixer: NLP-driven 'Auto-Heal' for linguistic repair.

The pipeline is entirely data-driven, utilizing settings defined in pyproject.toml.

 35    def _load_project_config(self) -> Dict[str, Any]:
 36        """Loads the configuration block from the TOML file."""
 37        if not self.config_path.exists():
 38            return {}
 39        try:
 40            with open(self.config_path, "rb") as f:
 41                return tomllib.load(f).get("tool", {}).get("transpiler-pro", {})
 42        except Exception:
 43            return {}

 45    def pre_process_markdown(self, content: str) -> str:
 46        """
 47        Shields Markdown blocks using patterns defined in configuration.
 48        
 49        Args:
 50            content: Raw Markdown string.
 51            
 52        Returns:
 53            Markdown string with complex blocks replaced by markers.
 54        """
 55        patterns = self.conv_cfg.get("shielding_patterns", [])
 56        
 57        for p in patterns:
 58            regex = p.get("regex")
 59            replacement = p.get("replacement")
 60            
 61            if p.get("hook") == "protect_spaces":
 62                def protect_hook(match: Match) -> str:
 63                    title = match.group(1).strip().replace(' ', 'PROTECT_SPACE')
 64                    body = match.group(2).strip()
 65                    return replacement.replace(r"\1", title).replace(r"\2", body)
 66                content = re.sub(regex, protect_hook, content, flags=re.S)
 67            else:
 68                content = re.sub(regex, replacement, content, flags=re.S)
 69        
 70        return content
 71
 72    def post_process_asciidoc(self, content: str) -> str:
 73        """
 74        Restores markers and cleans artifacts based on strict TOML rules.
 75        
 76        Args:
 77            content: Raw AsciiDoc produced by the transpiler.
 78            
 79        Returns:
 80            Finalized AsciiDoc with native syntax restored.
 81        """
 82        # 1. Generic Cleanup (e.g., removing frontmatter artifacts)
 83        cleanup = self.conv_cfg.get("cleanup_regex", [])
 84        for c in cleanup:
 85            flags = re.M if c.get("flags") == "M" else 0
 86            content = re.sub(c["regex"], c["replacement"], content, flags=flags)
 87
 88        # 2. Dynamic Marker Restoration
 89        restorations = self.conv_cfg.get("restoration_patterns", [])
 90        for r in restorations:
 91            regex = r.get("regex")
 92            replacement = r.get("replacement")
 93            
 94            if r.get("hook") == "restore_spaces":
 95                def restore_hook(match: Match) -> str:
 96                    title = match.group(1).replace('PROTECT_SPACE', ' ').strip()
 97                    body = match.group(2).strip()
 98                    return replacement.replace(r"\1", title).replace(r"\2", body)
 99                content = re.sub(regex, restore_hook, content, flags=re.S)
100            else:
101                mapping = r.get("map")
102                if mapping:
103                    for key, val in mapping.items():
104                        current_regex = regex.replace("{key}", key)
105                        current_replace = replacement.replace("{val}", val)
106                        content = re.sub(current_regex, current_replace, content, flags=re.S)
107                else:
108                    content = re.sub(regex, replacement, content, flags=re.S)
109
110        # 3. Dynamic XREFs and Extension Mapping
111        ext_map = self.conv_cfg.get("extension_map", {})
112        if ext_map:
113            normalization = self.conv_cfg.get("path_normalization", [])
114            
115            def clean_xref(match: Match) -> str:
116                path, ext = match.group(1), match.group(2)
117                for rule in normalization:
118                    path = re.sub(rule["regex"], rule["replacement"], path)
119                
120                new_ext = ext_map.get(ext, ext)
121                return f'xref:{path}.{new_ext}'
122            
123            xref_pattern = self.conv_cfg.get("xref_detection_regex", r'link:((?!http)[^ ]*)\.(md|json|yaml|yml)')
124            content = re.sub(xref_pattern, clean_xref, content)
125
126        return content.strip()

128    def convert_file(self, input_path: Path, output_path: Path) -> None:
129        """
130        Orchestrates the conversion of a single file.
131        
132        Uses Pandoc for the base conversion from Markdown to AsciiDoc.
133        """
134        raw_md = input_path.read_text(encoding='utf-8')
135        ready_md = self.pre_process_markdown(raw_md)
136        
137        temp_md = input_path.with_suffix('.tmp.md')
138        temp_md.write_text(ready_md)
139        
140        try:
141            # PANDOC INTEGRATION: Replaces kramdoc
142            # --ascii is used to ensure high compatibility with Antora/AsciiDoc
143            subprocess.run(
144                [
145                    "pandoc", 
146                    "-f", "markdown", 
147                    "-t", "asciidoc", 
148                    "--wrap=none", 
149                    "-o", str(output_path), 
150                    str(temp_md)
151                ], 
152                check=True, 
153                capture_output=True
154            )
155            
156            final_adoc = self.post_process_asciidoc(output_path.read_text())
157            output_path.write_text(final_adoc)
158        finally:
159            if temp_md.exists(): 
160                temp_md.unlink()

 26    """
 27    Orchestrates style validation using externalized configurations and Vale.
 28    
 29    Attributes:

 31        config_path (Path): Path to the project configuration.
 32        vale_ini (Path): The temporary configuration file generated for Vale.

 34
 35    def __init__(self, target_path: Path, config_path: Optional[Path] = None):
 36        """Initializes the linter with path isolation for configuration."""
 37        self.target_path = target_path
 38        self.config_path = config_path or Path("pyproject.toml")
 39        
 40        # Isolation: .vale.ini generated in context of current config
 41        self.vale_ini: Path = self.config_path.parent / ".vale.ini"
 42        
 43        self.config = self._load_project_config()
 44        self.guide_url = self.config.get("meta", {}).get("guide_url", "")

 46    def _load_project_config(self) -> Dict[str, Any]:
 47        """Loads linter-specific metadata from the dynamic config path."""
 48        if not self.config_path.exists():
 49            return {}
 50        try:
 51            with open(self.config_path, "rb") as f:
 52                return tomllib.load(f).get("tool", {}).get("transpiler-pro", {})
 53        except Exception:
 54            return {}

 56    def setup_config(self) -> None:
 57        """
 58        Generates a `.vale.ini` file using dynamic styles and levels from TOML.
 59        
 60        This allows the linter to adapt to different style guides (like SUSE) 
 61        without hardcoded paths.
 62        """
 63        linter_cfg = self.config.get("linter", {})
 64        styles_root = str(STYLES_DIR.resolve()).replace("\\", "/")
 65        
 66        styles = linter_cfg.get("styles", [])
 67        styles_str = ", ".join(styles) if styles else "Vale"
 68        min_level = linter_cfg.get("min_alert_level", "suggestion")
 69
 70        config_raw = f"""
 71        StylesPath = {styles_root}
 72        MinAlertLevel = {min_level}
 73
 74        [*.{{adoc,md}}]
 75        BasedOnStyles = {styles_str}
 76        """
 77        
 78        self.vale_ini.write_text(textwrap.dedent(config_raw).strip())
 79
 80    def _extract_suggestion(self, issue: Dict[str, Any]) -> str:
 81        """Extracts a repair suggestion from a Vale issue using TOML-defined patterns."""
 82        action_params = issue.get("Action", {}).get("Params", [])
 83        patterns_cfg = self.config.get("patterns", {})
 84        ignored = patterns_cfg.get("ignored_placeholders", [])
 85        
 86        if action_params:
 87            candidate = str(action_params[0])
 88            if candidate not in ignored:
 89                return candidate
 90
 91        search_pool = issue.get("Description", "") + " " + issue.get("Message", "")
 92        pattern = patterns_cfg.get("suggestion_extraction")
 93        
 94        if pattern and search_pool.strip():
 95            match = re.search(pattern, search_pool)
 96            if match:
 97                return match.group(1)
 98        
 99        return ""

101    def run(self) -> Dict[str, List[Dict[str, Any]]]:
102        """
103        Executes the Vale CLI and processes findings into a standardized format.
104        
105        Returns:
106            A dictionary where keys are file paths and values are lists of issues.
107        """
108        try:
109            abs_target = str(self.target_path.resolve())
110            
111            result = subprocess.run(
112                ["vale", "--config", str(self.vale_ini.resolve()), "--output=JSON", abs_target],
113                capture_output=True,
114                text=True,
115                check=False
116            )
117            
118            if not result.stdout or result.stdout.strip() == "":
119                return {}
120
121            raw_data = json.loads(result.stdout)
122            processed_findings = {}
123
124            for file_path, file_issues in raw_data.items():
125                processed_findings[file_path] = []
126                for issue in file_issues:
127                    processed_findings[file_path].append({
128                        "Line": issue.get("Line"),
129                        "Check": issue.get("Check"),
130                        "Severity": issue.get("Severity"),
131                        "Message": issue.get("Message"),
132                        "Description": issue.get("Description", ""),
133                        "Suggestion": self._extract_suggestion(issue)
134                    })
135                
136            return processed_findings
137
138        except (FileNotFoundError, json.JSONDecodeError, subprocess.SubprocessError):
139            return {}

152        })
153
154        table = Table(title="Style Guide Validation Report", title_style="bold cyan")

156        table.add_column("Severity", style="bold")

158        table.add_column("Rule ID", style="yellow")

 77    def _get_progressive_verb(self, verb_token) -> str:
 78        """
 79        Converts a verb to its progressive (-ing) form.
 80        
 81        Prioritizes the `[tool.transpiler-pro.grammar.special_verbs]` table from 
 82        pyproject.toml to ensure precision over algorithmic guessing.
 83        """
 84        lemma = verb_token.lemma_.lower()
 85        grammar_cfg = self.config.get("grammar", {})
 86        special = grammar_cfg.get("special_verbs", {})
 87        
 88        if lemma in special:
 89            return special[lemma]
 90
 91        # Algorithmic fallbacks
 92        if lemma.endswith("e") and not lemma.endswith("ee"):
 93            return lemma[:-1] + "ing"
 94        if len(lemma) > 2 and lemma[-1] not in "aeiou" and lemma[-2] in "aeiou" and lemma[-3] not in "aeiou":
 95            return lemma + lemma[-1] + "ing"
 96        return lemma + "ing"
 97
 98    def _fix_tense(self, line: str) -> str:
 99        """
100        Shifts sentence tense from future ('will') to progressive present.
101        
102        Example: "We will test" -> "We are testing".
103        """
104        if not self.nlp: 
105            return line
106        doc = self.nlp(line)
107        working_line = line
108        for token in doc:
109            if token.text.lower() == "will":
110                main_verb = token.head
111                if main_verb.pos_ == "VERB":
112                    subjects = [w for w in main_verb.lefts if "subj" in w.dep_]
113                    is_plural = any("Number=Plur" in str(s.morph) or s.text.lower() in ["we", "they", "you"] for s in subjects)
114                    aux = "are" if is_plural else "is"
115                    prog = self._get_progressive_verb(main_verb)
116                    working_line = re.sub(rf"\b{token.text}\s+{main_verb.text}\b", f"{aux} {prog}", working_line, flags=re.IGNORECASE)
117        return working_line

119    def fix_file(self, file_path: Path, violations: List[Dict[str, Any]]) -> int:
120        """
121        Orchestrates surgical repairs and global enforcement on a single file.
122        
123        Args:
124            file_path: Path to the AsciiDoc output.
125            violations: List of issues identified by the StyleLinter.
126            
127        Returns:
128            int: Total number of lines modified.
129        """
130        if not file_path.exists(): 
131            return 0
132        content = file_path.read_text(encoding="utf-8").splitlines()
133        total_fixes = 0
134        
135        line_map = defaultdict(list)
136        for v in violations: 
137            line_map[v.get("Line", 0)].append(v)
138
139        patterns = self.config.get("patterns", {})
140        extract_re = patterns.get("suggestion_extraction", r"'(.*?)'")
141        remove_trigger = patterns.get("removal_trigger", "removing")
142
143        # Combine branding and learned words for high-priority matching
144        session_branding = {**self.kb.get("learned", {}), **self.kb.get("branding", {})}
145
146        for line_num in sorted(line_map.keys(), reverse=True):
147            idx = line_num - 1
148            if idx < 0 or idx >= len(content): continue
149            
150            working_line = content[idx]
151            original_line = working_line
152
153            # --- PHASE A: LINTER-DRIVEN REPAIRS ---
154            for issue in line_map[line_num]:
155                msg = issue.get("Message", "")
156                check_id = issue.get("Check", "")
157
158                # 1. Branding Sync
159                for wrong, correct in session_branding.items():
160                    if f"'{wrong}'" in msg.lower() or f"{wrong}" in msg.lower():
161                        working_line = re.sub(rf"\b{re.escape(wrong)}\b", correct, working_line, flags=re.IGNORECASE)
162
163                # 2. Surgical Removal
164                if remove_trigger in msg.lower():
165                    targets = re.findall(extract_re, msg)
166                    if targets:
167                        working_line = re.sub(rf"\b{re.escape(targets[0])}\b\s?", "", working_line, flags=re.IGNORECASE)
168
169                # 3. Phrasal Substitution
170                elif "instead of" in msg.lower():
171                    terms = re.findall(extract_re, msg)
172                    if len(terms) >= 2:
173                        working_line = re.sub(rf"\b{re.escape(terms[1])}\b", terms[0], working_line, flags=re.IGNORECASE)
174
175                # 4. Spelling + Learning Discovery
176                elif "Spelling" in check_id:
177                    targets = re.findall(extract_re, msg)
178                    if targets:
179                        word = targets[0]
180                        if word.lower() not in session_branding:
181                            capitalized = word.capitalize()
182                            working_line = re.sub(rf"\b{re.escape(word)}\b", capitalized, working_line, flags=re.IGNORECASE)
183                            if word.lower() not in self.kb["branding"]:
184                                self.kb["learned"][word.lower()] = capitalized
185
186                # 5. Tense Shift
187                if check_id == "common.Will":
188                    working_line = self._fix_tense(working_line)
189
190            # --- PHASE B: GLOBAL ENFORCEMENT PASS ---
191            for wrong, correct in self.kb.get("branding", {}).items():
192                if re.search(rf"\b{re.escape(wrong)}\b", working_line, flags=re.IGNORECASE):
193                    working_line = re.sub(rf"\b{re.escape(wrong)}\b", correct, working_line)
194
195            if working_line != original_line:
196                content[idx] = working_line
197                total_fixes += 1
198
199        file_path.write_text("\n".join(content), encoding="utf-8")
200        self._save_kb() 
201        return total_fixes
