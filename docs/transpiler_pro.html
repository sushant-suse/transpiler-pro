 15    """A deterministic conversion engine that transforms Markdown into AsciiDoc.
 16    
 17    The engine uses a 'Marker Strategy' to shield specific Markdown components 
 18    (like admonitions and collapsible blocks) that are typically mishandled by 
 19    standard conversion tools.
 20    """
 21
 22    def __init__(self):
 23        # Map of Markdown-style admonition types to AsciiDoc-standard types
 24        self.ad_map: Dict[str, str] = {
 25            'note': 'NOTE', 
 26            'info': 'IMPORTANT', 
 27            'tip': 'TIP', 
 28            'warning': 'CAUTION'
 29        }

 31    def pre_process_markdown(self, content: str) -> str:
 32        """Shields complex Markdown elements before the primary transpilation phase.
 33        
 34        Args:
 35            content (str): The raw Markdown input.
 36            
 37        Returns:
 38            str: Markdown content with complex blocks replaced by stable markers.
 39        """
 40        # 1. Protect Admonitions (:::type ... :::)
 41        # We wrap these in unique markers to prevent line-merging during transpilation
 42        ad_pattern = r':::(?P<type>\w+)\n(?P<body>.*?)\n:::'
 43        content = re.sub(
 44            ad_pattern, 
 45            r'MARKER_ADMON_START_\1\n\2\nMARKER_ADMON_END', 
 46            content, 
 47            flags=re.S
 48        )
 49        
 50        # 2. Protect Collapsible Details (<details><summary>...</summary>...</details>)
 51        # We glue the summary text using 'PROTECT_SPACE' to prevent kramdoc from wrapping titles
 52        def protect_summary(match: Match) -> str:
 53            title = match.group('title').strip().replace(' ', 'PROTECT_SPACE')
 54            body = match.group('body').strip()
 55            return f"MARKER_COLL_START_{title}\n{body}\nMARKER_COLL_END"
 56
 57        coll_pattern = r'<details>\s*<summary>(?P<title>.*?)</summary>\s*(?P<body>.*?)\s*</details>'
 58        content = re.sub(coll_pattern, protect_summary, content, flags=re.S)
 59        
 60        return content
 61
 62    def post_process_asciidoc(self, content: str) -> str:
 63        """Finalizes the AsciiDoc structure by restoring markers and refining syntax.
 64        
 65        Args:
 66            content (str): The raw AsciiDoc output from kramdoc.
 67            
 68        Returns:
 69            str: Polished, Antora-compliant AsciiDoc content.
 70        """
 71        # 1. Clean up Metadata: Remove kramdoc-specific headers and auto-generated IDs
 72        content = re.sub(r'^:.*?\n', '', content, flags=re.M)
 73        content = re.sub(r'\[#.*?\]\n', '', content)
 74        
 75        # 2. Restore Admonitions: Reconstruct markers into standard [TYPE] blocks
 76        for md_type, ad_type in self.ad_map.items():
 77            pattern = rf'MARKER_ADMON_START_{md_type}\s*(.*?)\s*MARKER_ADMON_END'
 78            content = re.sub(pattern, rf'[{ad_type}]\n====\n\1\n====', content, flags=re.S)
 79
 80        # 3. Handle Legacy Notes: Convert blockquote notes (e.g., '> Note:') into formal blocks
 81        content = re.sub(r'____\n\*Note\*:\s*(.*?)\n____', r'[NOTE]\n====\n\1\n====', content, flags=re.S)
 82        content = re.sub(r'^NOTE:\s*(.*)$', r'[NOTE]\n====\n\1\n====', content, flags=re.M)
 83
 84        # 4. Refine Navigation: Convert internal links into Antora XREFs
 85        def clean_xref(match: Match) -> str:
 86            path = match.group(1)
 87            ext = match.group(2)
 88            # Normalize path: remove redundant './' prefixes
 89            path = re.sub(r'^\.\/', '', path)
 90            # Map .md extensions to .adoc
 91            new_ext = 'adoc' if ext == 'md' else ext
 92            return f'xref:{path}.{new_ext}'
 93
 94        # Match relative links while ignoring absolute 'http' URLs
 95        content = re.sub(r'link:((?!http)[^ ]*)\.(md|json|yaml|yml)', clean_xref, content)
 96
 97        # 5. Correct List Depth: Fix nested numbering level in mixed bullet/number lists
 98        content = re.sub(r'(\* .*?\n)\. ', r'\1.. ', content)
 99
100        # 6. Admonition Safety: Convert internal headers to bold to avoid syntax collision
101        def fix_internal_headings(match: Match) -> str:
102            header, body = match.group(1), match.group(2)
103            fixed_body = re.sub(r'^={1,6}\s+(.*)$', r'*\1*', body, flags=re.M)
104            return f"{header}\n====\n{fixed_body}\n===="
105        content = re.sub(r'(\[(?:IMPORTANT|NOTE|TIP|CAUTION)\])\n====\n(.*?)\n====', fix_internal_headings, content, flags=re.S)
106
107        # 7. Restore Collapsible Blocks: Rebuild [%collapsible] sections
108        pattern_coll = r'MARKER_COLL_START_(.*?)\s+(.*?)\s*MARKER_COLL_END'
109        def restore_collapsible(match: Match) -> str:
110            title = match.group(1).replace('PROTECT_SPACE', ' ').strip()
111            body = match.group(2).strip()
112            return f".{title}\n[%collapsible]\n======\n{body}\n======"
113        content = re.sub(pattern_coll, restore_collapsible, content, flags=re.S)
114
115        # 8. Structural Cleanup: Remove excess whitespace for clean output
116        content = re.sub(r'(\[.*?\])\n\n+(====|======)', r'\1\n\2', content)
117        content = re.sub(r'\n{3,}', '\n\n', content)
118        
119        return content.strip()

121    def convert_file(self, input_path: Path, output_path: Path) -> None:
122        """Executes the full conversion pipeline for a single file.
123
124        Args:
125            input_path (Path): The path to the source Markdown file.
126            output_path (Path): The path where the converted AsciiDoc will be saved.
127            
128        Raises:
129            subprocess.CalledProcessError: If the underlying kramdoc execution fails.
130            IOError: If there are issues reading or writing the files.
131        """
132        # Load raw content
133        raw_md = input_path.read_text(encoding='utf-8')
134        
135        # Phase 1: Pre-process
136        ready_md = self.pre_process_markdown(raw_md)
137        temp_md = input_path.with_suffix('.tmp.md')
138        temp_md.write_text(ready_md)
139
140        try:
141            # Phase 2: Core Transpilation
142            # We use '--wrap ventilate' to maintain logical line breaks for better regex matching
143            subprocess.run([
144                "kramdoc", "--wrap", "ventilate", "--auto-ids",
145                "-o", str(output_path), str(temp_md)
146            ], check=True)
147            
148            # Phase 3: Post-process and cleanup
149            adoc_output = output_path.read_text()
150            final_adoc = self.post_process_asciidoc(adoc_output)
151            output_path.write_text(final_adoc)
152            
153        finally:
154            # Always remove the temporary shielded file
155            if temp_md.exists():
156                temp_md.unlink()

 21class StyleLinter:
 22    """Orchestrates the style validation phase using the Vale linting engine.
 23    
 24    This class manages the lifecycle of the linter configuration, executes 
 25    rule-based checks against converted documents, and generates visual reports.
 26    """
 27
 28    def __init__(self, target_path: Path):
 29        """Initializes the linter for a specific document.
 30        
 31        Args:
 32            target_path (Path): Path to the .adoc or .md file to be validated.
 33        """
 34        self.target_path = target_path
 35        self.vale_ini: Path = BASE_DIR / ".vale.ini"
 36        self.guide_url: str = "https://documentation.suse.com/style/current/html/style-guide-adoc/index.html"
 37
 38    def setup_config(self) -> None:
 39        """Dynamically generates a Vale configuration file (.vale.ini) tailored
 40        to the project's style requirements and local paths.
 41        """
 42        styles_root = STYLES_DIR 
 43        
 44        # Maintenance: Remove conflicting default spelling rules to prioritize 
 45        # the specialized SUSE dictionary.
 46        spelling_file = styles_root / "common" / "Spelling.yml"
 47        if spelling_file.exists():
 48            spelling_file.unlink()
 49
 50        # Define the linting environment
 51        # 'config', 'common', and 'asciidoc' refer to the SUSE-specific YAML rulesets.
 52        config_raw = f"""
 53        StylesPath = {styles_root}
 54        MinAlertLevel = suggestion
 55
 56        [*.{{adoc,md}}]
 57        BasedOnStyles = Vale, config, common, asciidoc
 58        """
 59        
 60        self.vale_ini.write_text(textwrap.dedent(config_raw).strip())
 61
 62    def run(self) -> Dict[str, List[Dict[str, Any]]]:
 63        """Executes the Vale CLI against the target document and parses the result.
 64
 65        Returns:
 66            Dict[str, List[Dict[str, Any]]]: A structured dictionary where keys 
 67                are file paths and values are lists of issue dictionaries 
 68                containing 'Line', 'Severity', 'Message', and 'Check'.
 69        """
 70        try:
 71            # Resolve absolute path to ensure Vale identifies the file correctly
 72            abs_target = self.target_path.resolve()
 73            
 74            # Execute Vale with JSON output format for programmatic parsing
 75            result = subprocess.run(
 76                ["vale", "--config", str(self.vale_ini), "--output=JSON", str(abs_target)],
 77                capture_output=True,
 78                text=True,
 79                check=False
 80            )
 81            
 82            if result.stderr:
 83                console.print(f"[yellow]Vale Execution Warning:[/] {result.stderr}")
 84                
 85            return json.loads(result.stdout) if result.stdout else {}
 86
 87        except FileNotFoundError:
 88            console.print("[bold red]Critical Error:[/] Vale CLI is not installed or not in PATH.")
 89            return {}
 90        except json.JSONDecodeError:
 91            console.print("[bold red]Analysis Error:[/] Could not interpret Vale JSON output.")
 92            return {}
 93
 94    def display_report(self, data: Dict[str, List[Dict[str, Any]]]) -> None:
 95        """Parses raw linting data and renders a formatted table in the terminal.
 96
 97        Args:
 98            data (Dict[str, List[Dict[str, Any]]]): The JSON findings returned from 
 99                the run() method, containing lists of style violations.
100        """
101        # Early exit if no issues are detected
102        if not data or not any(data.values()):
103            console.print("\n [bold green]Quality Check Passed:[/] No style violations detected.")
104            return
105
106        table = Table(title="Style Guide Validation Report", title_style="bold cyan")
107        table.add_column("Line", style="magenta", justify="right")
108        table.add_column("Severity", style="bold")
109        table.add_column("Message", style="white")
110        table.add_column("Rule ID", style="yellow")
111
112        # Iterate through findings for the specific file
113        for _, issues in data.items():
114            for issue in issues:
115                severity = issue['Severity']
116                
117                # Assign semantic colors based on violation level
118                color = {
119                    "error": "red",
120                    "warning": "yellow",
121                    "suggestion": "blue"
122                }.get(severity, "white")
123                
124                table.add_row(
125                    str(issue['Line']),
126                    f"[{color}]{severity}[/]",
127                    issue['Message'],
128                    issue['Check']
129                )
130
131        console.print(table)
132        console.print(f"\n [dim]Reference:[/] [link={self.guide_url}]Official SUSE Style Guide[/link]\n")
